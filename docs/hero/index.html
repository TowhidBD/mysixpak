HTML Structure#

You need to follow the below structure.

<!-- Hero Area Start -->
<section class="hero-area" id="Top">
<!-- Large devices Header start -->
<header class="header large-d-menu">...</header>
<!-- large device Header End -->
<!-- Small devices Header start -->
<header class="small-d-menu hs-menubar">...</header>
<!-- Small devices Header end -->
<!-- Small devices nav menu start-->
<nav class="hs-navigation">...</nav>
<!-- Small devices nav menu end -->
<!-- Carousel Start -->
<div class="hero-carousel">
<div class="owl-carousel owl-theme">
...
</div>
<!-- Carousel End -->
<!-- Carousel Nav Start -->
<div class="owl-nav-wrap">
...
</div>
<!-- Carousel Nav End -->
<!-- Carousel Dots Start -->
<div class="owl-dots-wrap">
...
</div>
<!-- Carousel Dots End -->
</div>
<div class="scroll-down-btn">
...
</div>
</section>
<!-- Hero Area End -->
